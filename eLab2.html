at_home_deliveries <- read_excel("D:/Facultate/Anul 2/Period 5/eLab II/Case 2/Input files/data_Maastricht_2023.xlsx", sheet = "At-home Deliveries")

self_pickedUp <- read_excel("D:/Facultate/Anul 2/Period 5/eLab II/Case 2/Input files/data_Maastricht_2023.xlsx", sheet = "Service Point Parcels Picked Up")
# plot amounts of At-Home delivers per week of the day

# Transform the data to long format
at_home_deliveries_Total <- at_home_deliveries %>%
  pivot_longer(
    cols = -c(day, Month, DayOfWeek), 
    names_to = "Location", 
    values_to = "Deliveries"
  )


at_home_deliveries_Total <- at_home_deliveries_Total %>%
  mutate(DayOfWeek = factor(DayOfWeek, levels = c("Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday")))


atHome_deliveries_summary <- at_home_deliveries_Total %>%
  group_by(DayOfWeek) %>%
  summarise(TotalDeliveries = sum(Deliveries))


ggplot(atHome_deliveries_summary, aes(x = DayOfWeek, y = TotalDeliveries, fill = DayOfWeek)) +
  geom_col() +
  labs(title = "Total At-Home Deliveries by Day of the Week",
       x = "Day of the Week",
       y = "Total Deliveries",
       fill = "Day of the Week") +
  theme_minimal()
